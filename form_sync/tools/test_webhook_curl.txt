# Quick Webhook Test Commands
# Copy and paste these commands into your terminal

# Basic test (no signature)
curl -X POST 'http://localhost:8080/form_sync/webhook/framer/Contact' \
  -H 'Content-Type: application/json' \
  -H 'Framer-Webhook-Submission-Id: test_1234567890_1234' \
  -d '{
    "name": "Test User",
    "email": "test@example.com",
    "phone": "+1234567890",
    "message": "Test webhook submission",
    "company": "Test Company Inc."
  }'

# Test with signature (replace YOUR_SECRET with your actual webhook secret)
# First, generate the signature:
# 1. Create payload: {"name":"Test User","email":"test@example.com"}
# 2. Concatenate: payload + submission_id
# 3. Generate HMAC: SHA-256(payload+submission_id, secret)
# 4. Format: sha256=<hex_hmac>
